using UnityEngine; public class ArabaKontrol : MonoBehaviour { // Arabanın tekerlek nesneleri (Unity içindeki 3D modeller) public WheelCollider onSol, onSag; public WheelCollider arkaSol, arkaSag; public float motorGucu = 1500f; public float donmeAcisi = 35f; void FixedUpdate() { // Klavyeden veya telefondan gelen girdileri al (Gaz ve Direksiyon) float gaz = Input.GetAxis("Vertical"); // W/S veya İleri/Geri float direksiyon = Input.GetAxis("Horizontal"); // A/D veya Sağ/Sol // Arka tekerleklere güç ver (Arkadan itişli) arkaSol.motorTorque = gaz * motorGucu; arkaSag.motorTorque = gaz * motorGucu; // Ön tekerlekleri döndür onSol.steerAngle = direksiyon * donmeAcisi; onSag.steerAngle = direksiyon * donmeAcisi; } }
